void loop()
{
  if(Serial.available() > 0)
  {
    String info;
    info=Serial.readStringUntil('\n');
    if(info.equals("ON"))
    {
     digitalWrite(ledPin,HIGH);
     Serial.println("LED is ON"); 
    }
    else if(info.equals("OFF"))
   {
     digitalWrite(ledPin,LOW);
     Serial.println("LED is OFF"); 
    }
   }
}
